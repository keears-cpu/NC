const SHEET_NAME = 'charts';
const SPREADSHEET_ID = '';
const HEADERS = [
  'created_at', 'updated_at', 'record_id', 'chart_id', 'engine_name', 'status',
  'source_type', 'extraction_mode', 'classification', 'notes',
  'person_name', 'phone', 'email', 'birth_date', 'birth_time_local', 'timezone',
  'birth_place_name', 'country_code', 'latitude', 'longitude',
  'zodiac_type', 'house_system', 'node_mode', 'lilith_mode', 'fortune_formula',
  'asc_formatted', 'mc_formatted', 'dsc_formatted', 'ic_formatted',
  'house_1', 'house_2', 'house_3', 'house_4', 'house_5', 'house_6',
  'house_7', 'house_8', 'house_9', 'house_10', 'house_11', 'house_12',
  'sun', 'moon', 'mercury', 'venus', 'mars', 'jupiter', 'saturn', 'uranus', 'neptune', 'pluto',
  'north_node_true', 'lilith_mean', 'chiron', 'ceres', 'pallas', 'juno', 'vesta', 'vertex', 'fortune',
  'core_complete', 'soft_missing_json', 'warnings_json', 'aspects_summary', 'provenance', 'chart_json',
  'report_preset', 'report_addons_json', 'report_id', 'viewer_code', 'product_code', 'payment_status',
  'payment_provider', 'payment_order_id', 'payment_transaction_id', 'payment_amount', 'payment_currency',
  'payment_requested_at', 'payment_approved_at', 'payment_webhook_status',
  'report_request_json', 'report_payload_json', 'report_html', 'report_html_url',
  'chart_svg', 'chart_svg_updated_at',
];

function doGet(e) {
  try {
    const action = String((e && e.parameter && e.parameter.action) || 'health').trim();
    if (action === 'list') {
      const limit = Number((e && e.parameter && e.parameter.limit) || 100) || 100;
      return jsonResponse_({ ok: true, sheet: SHEET_NAME, items: listRows_(limit) });
    }
    if (action === 'get') {
      const recordId = String((e && e.parameter && e.parameter.record_id) || '').trim();
      if (!recordId) return jsonResponse_({ ok: false, error: 'record_id_required' });
      const item = getRowByRecordId_(recordId);
      return jsonResponse_(item ? { ok: true, item } : { ok: false, error: 'record_not_found', record_id: recordId });
    }
    return jsonResponse_({ ok: true, sheet: SHEET_NAME, headers: HEADERS, message: 'charts storage ready' });
  } catch (error) {
    return jsonResponse_({ ok: false, error: String(error) });
  }
}

function doPost(e) {
  try {
    const payload = parsePostBody_(e);
    if (!payload.record_id && !payload.chart_id) {
      return jsonResponse_({ ok: false, error: 'record_id_required' });
    }

    const sheet = getSheet_();
    ensureHeaders_(sheet);
    const headerMap = getHeaderMap_(sheet);
    const recordId = String(payload.record_id || payload.chart_id || '').trim();
    const targetRow = findRowNumberByRecordId_(sheet, headerMap, recordId);
    const existing = targetRow > 1
      ? rowToObject_(sheet.getRange(targetRow, 1, 1, Math.max(sheet.getLastColumn(), HEADERS.length)).getValues()[0])
      : {};
    const rowObject = buildRowObject_(payload, existing);
    const normalizedRecordId = String(rowObject.record_id || '').trim();
    const rowValues = HEADERS.map((header) => rowObject[header] != null ? rowObject[header] : '');

    let rowUpdated = false;
    let rowAppended = false;
    let rowNumber = targetRow;

    if (targetRow > 1) {
      sheet.getRange(targetRow, 1, 1, HEADERS.length).setValues([rowValues]);
      rowUpdated = true;
    } else {
      sheet.appendRow(rowValues);
      rowNumber = sheet.getLastRow();
      rowAppended = true;
    }

    return jsonResponse_({
      ok: true,
      sheet: SHEET_NAME,
      row_number: rowNumber,
      record_id: normalizedRecordId,
      row_updated: rowUpdated,
      row_appended: rowAppended,
    });
  } catch (error) {
    return jsonResponse_({ ok: false, error: String(error) });
  }
}

function parsePostBody_(e) {
  const raw = e && e.postData && e.postData.contents ? e.postData.contents : '{}';
  return JSON.parse(raw);
}

function getSpreadsheet_() {
  if (SPREADSHEET_ID) return SpreadsheetApp.openById(SPREADSHEET_ID);
  return SpreadsheetApp.getActiveSpreadsheet();
}

function getSheet_() {
  const spreadsheet = getSpreadsheet_();
  let sheet = spreadsheet.getSheetByName(SHEET_NAME);
  if (!sheet) sheet = spreadsheet.insertSheet(SHEET_NAME);
  return sheet;
}

function ensureHeaders_(sheet) {
  const lastColumn = sheet.getLastColumn();
  if (lastColumn === 0) {
    sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]);
    return;
  }

  const existing = sheet.getRange(1, 1, 1, lastColumn).getValues()[0];
  const merged = existing.slice();
  let changed = false;

  for (var i = 0; i < HEADERS.length; i += 1) {
    if (existing[i] == null || existing[i] === '') {
      merged[i] = HEADERS[i];
      changed = true;
    }
  }

  if (merged.length < HEADERS.length) {
    for (var j = merged.length; j < HEADERS.length; j += 1) {
      merged[j] = HEADERS[j];
      changed = true;
    }
  }

  if (changed) {
    sheet.getRange(1, 1, 1, merged.length).setValues([merged]);
  }
}

function getHeaderMap_(sheet) {
  const headers = sheet.getRange(1, 1, 1, Math.max(sheet.getLastColumn(), HEADERS.length)).getValues()[0];
  const map = {};
  headers.forEach((header, index) => {
    if (header) map[String(header)] = index + 1;
  });
  return map;
}

function findRowNumberByRecordId_(sheet, headerMap, recordId) {
  if (!recordId || sheet.getLastRow() < 2) return 0;
  const column = headerMap.record_id;
  if (!column) return 0;
  const values = sheet.getRange(2, column, sheet.getLastRow() - 1, 1).getValues();
  for (let index = 0; index < values.length; index += 1) {
    if (String(values[index][0] || '').trim() === recordId) return index + 2;
  }
  return 0;
}

function listRows_(limit) {
  const sheet = getSheet_();
  ensureHeaders_(sheet);
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return [];
  const values = sheet.getRange(2, 1, lastRow - 1, Math.max(sheet.getLastColumn(), HEADERS.length)).getValues();
  const rows = values
    .map((row) => rowToObject_(row))
    .filter((item) => String(item.record_id || '').trim());
  rows.sort((a, b) => String(b.updated_at || b.created_at || '').localeCompare(String(a.updated_at || a.created_at || '')));
  return rows.slice(0, Math.max(1, limit));
}

function getRowByRecordId_(recordId) {
  const sheet = getSheet_();
  ensureHeaders_(sheet);
  const headerMap = getHeaderMap_(sheet);
  const rowNumber = findRowNumberByRecordId_(sheet, headerMap, recordId);
  if (!rowNumber) return null;
  const row = sheet.getRange(rowNumber, 1, 1, HEADERS.length).getValues()[0];
  return rowToObject_(row);
}

function rowToObject_(row) {
  const object = {};
  HEADERS.forEach((header, index) => {
    object[header] = row[index];
  });
  return object;
}

function buildRowObject_(payload, existing) {
  const now = new Date().toISOString();
  existing = existing || {};
  const chart = normalizeChart_(payload);
  const points = indexById_((chart && chart.points) || []);
  const bodies = indexById_((chart && chart.bodies) || []);
  const houses = indexByHouse_((chart && chart.houses) || []);
  const angles = (chart && chart.angles) || {};

  const row = {
    created_at: String(payload.created_at || existing.created_at || now),
    updated_at: now,
    record_id: String(payload.record_id || payload.chart_id || ''),
    chart_id: String(payload.chart_id || payload.record_id || ''),
    engine_name: String(payload.engine_name || get_(chart, ['metadata', 'engine_name']) || ''),
    status: String(payload.status || get_(chart, ['metadata', 'status']) || ''),
    source_type: String(payload.source_type || 'computed_by_fastapi'),
    extraction_mode: String(payload.extraction_mode || 'computed_by_swiss_ephemeris'),
    classification: String(payload.classification || 'computed_chart'),
    notes: String(payload.notes || ''),
    person_name: String(payload.person_name || get_(chart, ['input', 'person_name']) || ''),
    phone: String(payload.phone || ''),
    email: String(payload.email || ''),
    birth_date: String(payload.birth_date || get_(chart, ['input', 'birth_date']) || ''),
    birth_time_local: String(payload.birth_time_local || get_(chart, ['input', 'birth_time_local']) || ''),
    timezone: String(payload.timezone || get_(chart, ['input', 'timezone']) || ''),
    birth_place_name: String(payload.birth_place_name || get_(chart, ['input', 'birth_place_name']) || ''),
    country_code: String(payload.country_code || get_(chart, ['input', 'country_code']) || ''),
    latitude: String(payload.latitude != null ? payload.latitude : get_(chart, ['input', 'latitude']) || ''),
    longitude: String(payload.longitude != null ? payload.longitude : get_(chart, ['input', 'longitude']) || ''),
    zodiac_type: String(payload.zodiac_type || get_(chart, ['settings', 'zodiac_type']) || ''),
    house_system: String(payload.house_system || get_(chart, ['settings', 'house_system']) || ''),
    node_mode: String(payload.node_mode || get_(chart, ['settings', 'node_mode']) || ''),
    lilith_mode: String(payload.lilith_mode || get_(chart, ['settings', 'lilith_mode']) || ''),
    fortune_formula: String(payload.fortune_formula || get_(chart, ['settings', 'fortune_formula']) || ''),
    asc_formatted: formatAngle_(angles.asc),
    mc_formatted: formatAngle_(angles.mc),
    dsc_formatted: formatAngle_(angles.dsc),
    ic_formatted: formatAngle_(angles.ic),
    house_1: formatHouse_(houses[1]),
    house_2: formatHouse_(houses[2]),
    house_3: formatHouse_(houses[3]),
    house_4: formatHouse_(houses[4]),
    house_5: formatHouse_(houses[5]),
    house_6: formatHouse_(houses[6]),
    house_7: formatHouse_(houses[7]),
    house_8: formatHouse_(houses[8]),
    house_9: formatHouse_(houses[9]),
    house_10: formatHouse_(houses[10]),
    house_11: formatHouse_(houses[11]),
    house_12: formatHouse_(houses[12]),
    sun: formatPoint_(bodies.sun),
    moon: formatPoint_(bodies.moon),
    mercury: formatPoint_(bodies.mercury),
    venus: formatPoint_(bodies.venus),
    mars: formatPoint_(bodies.mars),
    jupiter: formatPoint_(bodies.jupiter),
    saturn: formatPoint_(bodies.saturn),
    uranus: formatPoint_(bodies.uranus),
    neptune: formatPoint_(bodies.neptune),
    pluto: formatPoint_(bodies.pluto),
    north_node_true: formatPoint_(points.north_node_true),
    lilith_mean: formatPoint_(points.lilith_mean),
    chiron: formatPoint_(points.chiron),
    ceres: formatPoint_(points.ceres),
    pallas: formatPoint_(points.pallas),
    juno: formatPoint_(points.juno),
    vesta: formatPoint_(points.vesta),
    vertex: formatPoint_(points.vertex),
    fortune: formatPoint_(points.fortune),
    core_complete: toSheetBoolean_(payload.core_complete != null ? payload.core_complete : get_(chart, ['availability', 'core_complete'])),
    soft_missing_json: stringifyJson_(payload.soft_missing_json, get_(chart, ['availability', 'soft_missing']) || []),
    warnings_json: stringifyJson_(payload.warnings_json, get_(chart, ['metadata', 'warnings']) || []),
    aspects_summary: buildAspectsSummary_(chart),
    provenance: stringifyJson_(payload.provenance, {
      source_type: String(payload.source_type || 'computed_by_fastapi'),
      extraction_mode: String(payload.extraction_mode || 'computed_by_swiss_ephemeris'),
      engine_name: String(payload.engine_name || get_(chart, ['metadata', 'engine_name']) || ''),
    }),
    chart_json: stringifyJson_(payload.chart_json, chart || payload.chart || {}),
    report_preset: String(payload.report_preset || existing.report_preset || ''),
    report_addons_json: stringifyJson_(payload.report_addons_json, payload.report_addons || parseJsonString_(existing.report_addons_json, [])),

    payment_status: String(payload.payment_status || payload.report_payment_status || existing.payment_status || ''),
    payment_provider: String(payload.payment_provider || existing.payment_provider || ''),
    payment_order_id: String(payload.payment_order_id || existing.payment_order_id || ''),
    payment_transaction_id: String(payload.payment_transaction_id || existing.payment_transaction_id || ''),
    payment_amount: String(payload.payment_amount != null ? payload.payment_amount : existing.payment_amount || ''),
    payment_currency: String(payload.payment_currency || existing.payment_currency || ''),
    payment_requested_at: String(payload.payment_requested_at || existing.payment_requested_at || ''),
    payment_approved_at: String(payload.payment_approved_at || existing.payment_approved_at || ''),
    payment_webhook_status: String(payload.payment_webhook_status || existing.payment_webhook_status || ''),
    report_request_json: stringifyJson_(payload.report_request_json, payload.report_request || parseJsonString_(existing.report_request_json, null)),
    report_payload_json: stringifyJson_(payload.report_payload_json, payload.report_payload || parseJsonString_(existing.report_payload_json, null)),
    report_html: String(payload.report_html || existing.report_html || ''),
    report_html_url: String(payload.report_html_url || existing.report_html_url || ''),
    chart_svg: String(payload.chart_svg || existing.chart_svg || ''),
    chart_svg_updated_at: String(payload.chart_svg_updated_at || existing.chart_svg_updated_at || ''),
  };

  HEADERS.forEach((header) => {
    if ((row[header] == null || row[header] === '') && existing[header] != null && existing[header] !== '') {
      row[header] = existing[header];
    }
  });

  return Object.assign(existing, row);
}

function normalizeChart_(payload) {
  if (payload.chart && typeof payload.chart === 'object') return payload.chart;
  if (payload.chart_json && typeof payload.chart_json === 'string') {
    try {
      return JSON.parse(payload.chart_json);
    } catch (error) {
      return null;
    }
  }
  return null;
}

function indexById_(items) {
  const output = {};
  (items || []).forEach((item) => {
    if (item && item.id) output[item.id] = item;
  });
  return output;
}

function indexByHouse_(items) {
  const output = {};
  (items || []).forEach((item) => {
    if (item && item.house != null) output[item.house] = item;
  });
  return output;
}

function formatAngle_(angle) {
  return angle && angle.formatted ? String(angle.formatted) : '';
}

function formatHouse_(house) {
  return house && house.formatted ? String(house.formatted) : '';
}

function formatPoint_(item) {
  if (!item) return '';
  const parts = [String(item.formatted || '')];
  if (item.retrograde) parts.push('Rx');
  if (item.house != null && item.house !== '') parts.push(String(item.house) + 'h');
  return parts.filter(Boolean).join(' ');
}

function buildAspectsSummary_(chart) {
  const aspects = (chart && chart.aspects) || [];
  return aspects.map((aspect) => {
    const pointA = String(aspect.point_a || '');
    const pointB = String(aspect.point_b || '');
    const type = String(aspect.aspect_type || '');
    const orb = String(aspect.orb_text || '');
    return [pointA, type, pointB, orb].filter(Boolean).join(' ');
  }).join(' | ');
}

function stringifyJson_(value, fallback) {
  if (typeof value === 'string' && value.trim()) return value;
  const source = value != null && value !== '' ? value : fallback;
  return JSON.stringify(source == null ? null : source);
}

function parseJsonString_(value, fallback) {
  if (value == null || value === '') return fallback;
  if (typeof value !== 'string') return value;
  try {
    return JSON.parse(value);
  } catch (error) {
    return fallback;
  }
}

function toSheetBoolean_(value) {
  if (typeof value === 'string') {
    return value.toLowerCase() === 'true' ? 'TRUE' : value.toLowerCase() === 'false' ? 'FALSE' : value;
  }
  return value ? 'TRUE' : 'FALSE';
}

function get_(source, path) {
  return path.reduce((acc, key) => (acc && acc[key] != null ? acc[key] : null), source);
}

function jsonResponse_(payload) {
  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}
