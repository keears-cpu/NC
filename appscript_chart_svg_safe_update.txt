function doGet(e) {
  try {
    const action = String(getParam_(e, 'action') || 'health').trim();

    if (action === 'auth_counselor') {
      const counselor = requireCounselorByCode_(String(getParam_(e, 'counselor_code') || '').trim());
      return jsonResponse_({ ok: true, counselor: publicCounselor_(counselor) });
    }

    if (action === 'list_counselors') {
      return jsonResponse_({
        ok: true,
        items: listCounselors_().map(publicCounselor_)
      });
    }

    if (action === 'list') {
      const limit = Number(getParam_(e, 'limit') || 100) || 100;
      const counselorCode = String(getParam_(e, 'counselor_code') || '').trim();
      const counselor = counselorCode ? requireCounselorByCode_(counselorCode) : null;
      const items = listRows_(limit, counselor);
      return jsonResponse_({
        ok: true,
        sheet: MASTER_SHEET_NAME,
        counselor: counselor ? publicCounselor_(counselor) : null,
        items: items
      });
    }

    if (action === 'get') {
      const recordId = String(getParam_(e, 'record_id') || '').trim();
      if (!recordId) return jsonResponse_({ ok: false, error: 'record_id_required' });
      const counselorCode = String(getParam_(e, 'counselor_code') || '').trim();
      const counselor = counselorCode ? requireCounselorByCode_(counselorCode) : null;
      const item = getRowByRecordId_(recordId, counselor);
      return jsonResponse_(
        item
          ? { ok: true, counselor: counselor ? publicCounselor_(counselor) : null, item: item }
          : { ok: false, error: 'record_not_found', record_id: recordId }
      );
    }

    return jsonResponse_({
      ok: true,
      sheet: MASTER_SHEET_NAME,
      chart_headers: CHART_HEADERS,
      counselor_headers: COUNSELOR_HEADERS,
      message: 'supabase_primary_sheet_backup_ready'
    });
  } catch (error) {
    return jsonResponse_({ ok: false, error: String(error) });
  }
}

function doPost(e) {
  try {
    const payload = parsePostBody_(e);
    const action = String(payload.action || '').trim();

    if (action === 'upsert_counselor') {
      const counselor = upsertCounselor_(payload);
      return jsonResponse_({ ok: true, counselor: publicCounselor_(counselor) });
    }

    if (!payload.record_id && !payload.chart_id) {
      return jsonResponse_({ ok: false, error: 'record_id_required' });
    }

    const counselor = resolveCounselorForWrite_(payload);
    enforceCounselorWriteRules_(payload, counselor);
    const rowObject = buildChartRowObject_(payload, counselor);
    const writeResult = upsertChartRow_(getChartSheet_(), rowObject);

    return jsonResponse_({
      ok: true,
      sheet: MASTER_SHEET_NAME,
      record_id: String(rowObject.record_id || ''),
      row_number: writeResult.rowNumber,
      row_updated: writeResult.rowUpdated,
      row_appended: writeResult.rowAppended,
      counselor: counselor ? publicCounselor_(counselor) : null
    });
  } catch (error) {
    return jsonResponse_({ ok: false, error: String(error) });
  }
}

function parsePostBody_(e) {
  const raw = e && e.postData && e.postData.contents ? e.postData.contents : '{}';
  return JSON.parse(raw);
}

function getParam_(e, key) {
  return e && e.parameter ? e.parameter[key] : '';
}

function getSpreadsheet_() {
  if (SPREADSHEET_ID) return SpreadsheetApp.openById(SPREADSHEET_ID);
  return SpreadsheetApp.getActiveSpreadsheet();
}

function getOrCreateSheetByName_(name) {
  const spreadsheet = getSpreadsheet_();
  let sheet = spreadsheet.getSheetByName(name);
  if (!sheet) sheet = spreadsheet.insertSheet(name);
  return sheet;
}

function ensureHeaders_(sheet, headers) {
  const lastColumn = sheet.getLastColumn();
  if (lastColumn === 0) {
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    return;
  }

  const existing = sheet.getRange(1, 1, 1, Math.max(lastColumn, headers.length)).getValues()[0];
  const merged = existing.slice();
  let changed = false;

  for (var i = 0; i < headers.length; i += 1) {
    if (existing[i] == null || existing[i] === '') {
      merged[i] = headers[i];
      changed = true;
    }
  }

  if (merged.length < headers.length) {
    for (var j = merged.length; j < headers.length; j += 1) {
      merged[j] = headers[j];
      changed = true;
    }
  }

  if (changed) {
    sheet.getRange(1, 1, 1, merged.length).setValues([merged]);
  }
}

function getHeaderMap_(sheet, headers) {
  const rowHeaders = sheet.getRange(1, 1, 1, Math.max(sheet.getLastColumn(), headers.length)).getValues()[0];
  const map = {};
  rowHeaders.forEach(function(header, index) {
    if (header) map[String(header)] = index + 1;
  });
  return map;
}

function getChartSheet_() {
  const sheet = getOrCreateSheetByName_(MASTER_SHEET_NAME);
  ensureHeaders_(sheet, CHART_HEADERS);
  return sheet;
}

function getCounselorSheet_() {
  const sheet = getOrCreateSheetByName_(COUNSELOR_SHEET_NAME);
  ensureHeaders_(sheet, COUNSELOR_HEADERS);
  ensureSeedCounselors_(sheet);
  return sheet;
}

function ensureSeedCounselors_(sheet) {
  const existing = listCounselorsRaw_(sheet);
  const hasMaster = existing.some(function(item) {
    return String(item.access_code || '').trim().toUpperCase() === MASTER_ACCESS_CODE;
  });
  const hasTest = existing.some(function(item) {
    return String(item.access_code || '').trim().toUpperCase() === TEST_ACCESS_CODE;
  });

  if (!hasMaster) {
    upsertCounselor_({
      name: 'Master Access',
      access_code: MASTER_ACCESS_CODE,
      status: 'active',
      usage_scope: 'all',
      can_write: 'TRUE',
      can_read_all: 'TRUE',
      max_records: '',
      allowed_presets: '*',
      admin_link: DEFAULT_ADMIN_LINK || '',
      notes: 'seed_master_access'
    });
  }

  if (!hasTest) {
    upsertCounselor_({
      name: '테스트 상담사',
      access_code: TEST_ACCESS_CODE,
      status: 'active',
      usage_scope: 'own_only',
      can_write: 'TRUE',
      can_read_all: 'FALSE',
      max_records: '3',
      allowed_presets: '*',
      admin_link: DEFAULT_ADMIN_LINK || '',
      notes: 'seed_test_access'
    });
  }
}

function listCounselors_() {
  return listCounselorsRaw_(getCounselorSheet_());
}

function listCounselorsRaw_(sheet) {
  if (sheet.getLastRow() < 2) return [];
  const values = sheet.getRange(2, 1, sheet.getLastRow() - 1, COUNSELOR_HEADERS.length).getValues();
  return values
    .map(function(row) { return rowToObject_(row, COUNSELOR_HEADERS); })
    .filter(function(item) { return String(item.access_code || '').trim(); });
}

function getCounselorByCode_(accessCode) {
  if (!accessCode) return null;
  const code = String(accessCode).trim().toUpperCase();
  const items = listCounselors_();
  for (var i = 0; i < items.length; i += 1) {
    if (String(items[i].access_code || '').trim().toUpperCase() === code) {
      return items[i];
    }
  }
  return null;
}

function requireCounselorByCode_(accessCode) {
  const counselor = getCounselorByCode_(accessCode);
  if (!counselor) throw new Error('counselor_not_found');
  if (!isCounselorActive_(counselor)) throw new Error('counselor_expired_or_inactive');
  return counselor;
}

function isCounselorActive_(counselor) {
  const status = String(counselor.status || 'active').trim().toLowerCase();
  if (status && ['active', 'trial', 'test'].indexOf(status) === -1) return false;

  const today = isoDate_(new Date());
  const startDate = normalizeDateValue_(counselor.start_date);
  const endDate = normalizeDateValue_(counselor.end_date);
  if (startDate && today < startDate) return false;
  if (endDate && today > endDate) return false;
  return true;
}

function publicCounselor_(counselor) {
  if (!counselor) return null;
  return {
    name: String(counselor.name || ''),
    accessCode: String(counselor.access_code || '').trim().toUpperCase(),
    startDate: normalizeDateValue_(counselor.start_date),
    endDate: normalizeDateValue_(counselor.end_date),
    status: String(counselor.status || 'active'),
    usageScope: String(counselor.usage_scope || 'own_only'),
    canWrite: parseSheetBoolean_(counselor.can_write, true),
    canReadAll: parseSheetBoolean_(counselor.can_read_all, false),
    maxRecords: parseNumber_(counselor.max_records),
    allowedPresets: parseAllowedPresets_(counselor.allowed_presets),
    adminLink: String(counselor.admin_link || ''),
    notes: String(counselor.notes || '')
  };
}

function resolveCounselorForWrite_(payload) {
  const directCode = String(payload.counselor_code || '').trim();
  if (!directCode) return null;
  return requireCounselorByCode_(directCode);
}

function enforceCounselorWriteRules_(payload, counselor) {
  if (!counselor) return;

  if (!parseSheetBoolean_(counselor.can_write, true)) {
    throw new Error('counselor_write_not_allowed');
  }

  const maxRecords = parseNumber_(counselor.max_records);
  if (maxRecords > 0) {
    const recordId = String(payload.record_id || payload.chart_id || '').trim();
    const existing = getRowByRecordId_(recordId, null);
    if (existing && String(existing.counselor_code || '').trim().toUpperCase() === String(counselor.access_code || '').trim().toUpperCase()) {
      return;
    }
    const currentCount = countRowsByCounselorCode_(String(counselor.access_code || ''));
    if (currentCount >= maxRecords) {
      throw new Error('counselor_max_records_reached');
    }
  }

  const allowedPresets = parseAllowedPresets_(counselor.allowed_presets);
  const incomingPreset = String(payload.report_preset || '').trim();
  if (incomingPreset && allowedPresets.length && allowedPresets[0] !== '*' && allowedPresets.indexOf(incomingPreset) === -1) {
    throw new Error('report_preset_not_allowed');
  }
}

function listRows_(limit, counselor) {
  const items = readRows_(counselor);
  items.sort(function(a, b) {
    return String(b.updated_at || b.created_at || '').localeCompare(String(a.updated_at || a.created_at || ''));
  });
  return items.slice(0, Math.max(1, limit));
}

function getRowByRecordId_(recordId, counselor) {
  const rows = readRows_(counselor);
  for (var i = 0; i < rows.length; i += 1) {
    if (String(rows[i].record_id || '').trim() === String(recordId || '').trim()) {
      return rows[i];
    }
  }
  return null;
}

function readRows_(counselor) {
  const sheet = getChartSheet_();
  if (sheet.getLastRow() < 2) return [];

  const values = sheet.getRange(2, 1, sheet.getLastRow() - 1, Math.max(sheet.getLastColumn(), CHART_HEADERS.length)).getValues();
  return values
    .map(function(row) { return rowToObject_(row, CHART_HEADERS); })
    .filter(function(item) {
      if (!String(item.record_id || '').trim()) return false;
      if (!counselor) return true;
      if (parseSheetBoolean_(counselor.can_read_all, false)) return true;
      return String(item.counselor_code || '').trim().toUpperCase() === String(counselor.access_code || '').trim().toUpperCase();
    });
}

function countRowsByCounselorCode_(counselorCode) {
  if (!counselorCode) return 0;
  const sheet = getChartSheet_();
  if (sheet.getLastRow() < 2) return 0;
  const headerMap = getHeaderMap_(sheet, CHART_HEADERS);
  const column = headerMap.counselor_code;
  if (!column) return 0;
  const values = sheet.getRange(2, column, sheet.getLastRow() - 1, 1).getValues();
  var count = 0;
  for (var i = 0; i < values.length; i += 1) {
    if (String(values[i][0] || '').trim().toUpperCase() === String(counselorCode).trim().toUpperCase()) {
      count += 1;
    }
  }
  return count;
}

function findRowNumberByRecordId_(sheet, headerMap, recordId) {
  if (!recordId || sheet.getLastRow() < 2) return 0;
  const column = headerMap.record_id;
  if (!column) return 0;
  const values = sheet.getRange(2, column, sheet.getLastRow() - 1, 1).getValues();
  for (var index = 0; index < values.length; index += 1) {
    if (String(values[index][0] || '').trim() === recordId) return index + 2;
  }
  return 0;
}

function upsertChartRow_(sheet, rowObject) {
  ensureHeaders_(sheet, CHART_HEADERS);
  const headerMap = getHeaderMap_(sheet, CHART_HEADERS);
  const recordId = String(rowObject.record_id || '').trim();
  const targetRow = findRowNumberByRecordId_(sheet, headerMap, recordId);
  const rowValues = CHART_HEADERS.map(function(header) {
    return rowObject[header] != null ? rowObject[header] : '';
  });

  if (targetRow > 1) {
    sheet.getRange(targetRow, 1, 1, CHART_HEADERS.length).setValues([rowValues]);
    return { rowNumber: targetRow, rowUpdated: true, rowAppended: false };
  }

  sheet.appendRow(rowValues);
  return { rowNumber: sheet.getLastRow(), rowUpdated: false, rowAppended: true };
}

function upsertCounselor_(payload) {
  const code = String(payload.access_code || '').trim().toUpperCase();
  if (!code) throw new Error('access_code_required');

  const sheet = getCounselorSheet_();
  const headerMap = getHeaderMap_(sheet, COUNSELOR_HEADERS);
  let rowNumber = 0;

  if (sheet.getLastRow() >= 2) {
    const values = sheet.getRange(2, headerMap.access_code, sheet.getLastRow() - 1, 1).getValues();
    for (var i = 0; i < values.length; i += 1) {
      if (String(values[i][0] || '').trim().toUpperCase() === code) {
        rowNumber = i + 2;
        break;
      }
    }
  }

  const existing = rowNumber > 1
    ? rowToObject_(sheet.getRange(rowNumber, 1, 1, COUNSELOR_HEADERS.length).getValues()[0], COUNSELOR_HEADERS)
    : {};
  const now = new Date().toISOString();
  const rowObject = {
    created_at: String(existing.created_at || payload.created_at || now),
    updated_at: now,
    name: String(payload.name || existing.name || ''),
    access_code: code,
    start_date: normalizeDateValue_(payload.start_date || existing.start_date || ''),
    end_date: normalizeDateValue_(payload.end_date || existing.end_date || ''),
    status: String(payload.status || existing.status || 'active'),
    usage_scope: String(payload.usage_scope || existing.usage_scope || (code === MASTER_ACCESS_CODE ? 'all' : 'own_only')),
    can_write: toSheetBoolean_(payload.can_write != null ? payload.can_write : parseSheetBoolean_(existing.can_write, true)),
    can_read_all: toSheetBoolean_(payload.can_read_all != null ? payload.can_read_all : code === MASTER_ACCESS_CODE ? true : parseSheetBoolean_(existing.can_read_all, false)),
    max_records: String(payload.max_records != null ? payload.max_records : existing.max_records || ''),
    allowed_presets: String(payload.allowed_presets || existing.allowed_presets || '*'),
    admin_link: String(payload.admin_link || existing.admin_link || DEFAULT_ADMIN_LINK || ''),
    notes: String(payload.notes || existing.notes || '')
  };

  const rowValues = COUNSELOR_HEADERS.map(function(header) {
    return rowObject[header] != null ? rowObject[header] : '';
  });

  if (rowNumber > 1) {
    sheet.getRange(rowNumber, 1, 1, COUNSELOR_HEADERS.length).setValues([rowValues]);
  } else {
    sheet.appendRow(rowValues);
    rowNumber = sheet.getLastRow();
  }

  return rowToObject_(sheet.getRange(rowNumber, 1, 1, COUNSELOR_HEADERS.length).getValues()[0], COUNSELOR_HEADERS);
}

function rowToObject_(row, headers) {
  const object = {};
  headers.forEach(function(header, index) {
    object[header] = row[index];
  });
  return object;
}

function stringifyJson_(directValue, fallbackValue) {
  var value = directValue != null ? directValue : fallbackValue;
  if (value == null || value === '') return '';
  if (typeof value === 'string') return value;
  return JSON.stringify(value);
}

function normalizeChart_(payload) {
  if (payload.chart && typeof payload.chart === 'object') return payload.chart;
  if (payload.chart_json && typeof payload.chart_json === 'string') {
    try {
      return JSON.parse(payload.chart_json);
    } catch (error) {
      return null;
    }
  }
  return null;
}

function indexById_(items) {
  var map = {};
  (items || []).forEach(function(item) {
    if (item && item.id) map[String(item.id)] = item;
  });
  return map;
}

function indexByHouse_(items) {
  var map = {};
  (items || []).forEach(function(item) {
    if (item && item.house != null) map[String(item.house)] = item;
  });
  return map;
}

function get_(object, path) {
  return path.reduce(function(acc, key) {
    return acc && acc[key] != null ? acc[key] : null;
  }, object);
}

function formatAngle_(angle) {
  if (!angle) return '';
  return [angle.sign, angle.formatted || '', angle.degree != null ? String(angle.degree) : ''].filter(Boolean).join(' ');
}

function formatHouse_(house) {
  if (!house) return '';
  return [house.sign, house.formatted || '', house.degree != null ? String(house.degree) : ''].filter(Boolean).join(' ');
}

function formatPoint_(point) {
  if (!point) return '';
  var parts = [point.sign || '', point.formatted || ''];
  if (point.house != null) parts.push('H' + point.house);
  if (point.retrograde) parts.push('R');
  return parts.filter(Boolean).join(' ');
}

function buildAspectsSummary_(chart) {
  var aspects = get_(chart, ['aspects']) || [];
  return aspects.map(function(item) {
    return [item.point_a, item.aspect_type, item.point_b, item.orb_text].filter(Boolean).join(' ');
  }).slice(0, 24).join(' | ');
}

function toSheetBoolean_(value) {
  if (value === true || value === false) return value ? 'TRUE' : 'FALSE';
  const text = String(value || '').trim().toUpperCase();
  if (text === 'TRUE' || text === 'FALSE') return text;
  return text;
}

function parseSheetBoolean_(value, fallbackValue) {
  const text = String(value == null ? '' : value).trim().toUpperCase();
  if (text === 'TRUE') return true;
  if (text === 'FALSE') return false;
  return fallbackValue;
}

function parseNumber_(value) {
  const number = Number(value);
  return isNaN(number) ? 0 : number;
}

function parseAllowedPresets_(value) {
  const text = String(value || '').trim();
  if (!text) return [];
  return text.split(',').map(function(item) { return item.trim(); }).filter(Boolean);
}

function normalizeDateValue_(value) {
  if (!value) return '';
  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value.getTime())) {
    return isoDate_(value);
  }
  var text = String(value).trim();
  var match = text.match(/^(\d{4}-\d{2}-\d{2})/);
  return match ? match[1] : text;
}

function isoDate_(date) {
  return Utilities.formatDate(date, Session.getScriptTimeZone(), 'yyyy-MM-dd');
}

function jsonResponse_(payload) {
  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}
